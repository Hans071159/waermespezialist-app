<script>
let resultData = [];
function calculate() {
const cost = parseFloat(document.getElementById("cost").value);
const type = document.getElementById("type").value;
const strompreis = parseFloat(document.getElementById("strompreis").value);
const gaspreis = parseFloat(document.getElementById("gaspreis").value);
const verbrauch = parseFloat(document.getElementById("verbrauch").value);
const cop = parseFloat(document.getElementById("cop").value);
const output = document.getElementById("output");
resultData = [];

if (isNaN(cost) || isNaN(strompreis) || isNaN(gaspreis) || isNaN(verbrauch) || isNaN(cop)) {
output.innerHTML = "<p style='color:red;'>Bitte alle Felder korrekt ausfüllen.</p>";
return;
}

const stromverbrauch = verbrauch / cop;
const stromkosten = stromverbrauch * strompreis;
const gaskosten = verbrauch * gaspreis;
const ersparnisJahr = gaskosten - stromkosten;
const monatsStrom = stromkosten / 12;
const monatsGas = gaskosten / 12;
const monatsErsparnis = monatsGas - monatsStrom;

output.innerHTML = `
<table style="width:100%; font-size:1em;">
<tr><td><strong>Monatliche Stromkosten:</strong></td><td style="text-align:right;">${monatsStrom.toFixed(2)} €</td></tr>
<tr><td><strong>Monatliche Gaskosten:</strong></td><td style="text-align:right;">${monatsGas.toFixed(2)} €</td></tr>
<tr><td><strong>Monatliche Ersparnis:</strong></td><td style="text-align:right; color:green;"><strong>${monatsErsparnis.toFixed(2)} €</strong></td></tr>
<tr><td colspan="2" style="height:15px;"></td></tr>
<tr><td><strong>Jährliche Ersparnis:</strong></td><td style="text-align:right; color:green;"><strong>${ersparnisJahr.toFixed(2)} €</strong></td></tr>
</table>
`;

let tableHTML = `<table><thead><tr><th>Förderstufe</th><th>Förderhöhe (€)</th><th>Eigenanteil (€)</th><th>Amortisation (Jahre)</th></tr></thead><tbody>`;
for (let p = 20; p <= 70; p += 10) {
let foerderung = type === "Warmwasser-WP" ? 0 : cost * (p / 100);
let eigenanteil = cost - foerderung;
let amortisation = ersparnisJahr > 0 ? (eigenanteil / ersparnisJahr).toFixed(1) : "nicht berechenbar";
tableHTML += `<tr><td>Bei ${p}%</td><td style="text-align:right;">${foerderung.toFixed(2)}</td><td style="text-align:right;">${eigenanteil.toFixed(2)}</td><td style="text-align:right;">${amortisation}</td></tr>`;
resultData.push([`Bei ${p}%`, `${foerderung.toFixed(2)}`, `${eigenanteil.toFixed(2)}`, `${amortisation}`, `${monatsStrom.toFixed(2)}`, `${monatsGas.toFixed(2)}`, `${monatsErsparnis.toFixed(2)}`, `${ersparnisJahr.toFixed(2)}`]);
}
tableHTML += "</tbody></table>";
output.innerHTML += tableHTML;
}

function openPDF() {
const { jsPDF } = window.jspdf;
const doc = new jsPDF();
doc.setFontSize(14);
doc.text("Ihr Ergebnis Wärmepumpen-Förderung & Amortisation", 14, 20);
doc.setFontSize(10);
doc.text("Hinweis: Die tatsächliche Förderhöhe ist abhängig von der Prüfung durch die KfW oder das zuständige BAFA.", 14, 30);
if (resultData.length > 0) {
doc.autoTable({
startY: 40,
head: [["Förderstufe", "Förderhöhe (€)", "Eigenanteil (€)", "Amortisation (Jahre)", "Monatliche Stromkosten (€)", "Monatliche Gaskosten (€)", "Monatliche Ersparnis (€)", "Jährliche Ersparnis (€)"]],
body: resultData,
styles: { halign: "right" },
headStyles: { halign: "center", fillColor: [0, 119, 182] },
});
} else {
doc.text("Bitte zuerst auf 'Berechne' klicken.", 14, 50);
}
const pdfData = doc.output('blob');
const url = URL.createObjectURL(pdfData);
window.open(url, '_blank');
}

if ('serviceWorker' in navigator) {
navigator.serviceWorker.register('service-worker.js').then(reg => console.log('Service Worker registriert!', reg)).catch(err => console.error('Service Worker Fehler:', err));
}
